export const SdkHeader = ({language, feature}) => (
<Note>
  <p>While each Modus SDK offers similar functionality, the APIs and usage may vary between languages.</p>
  <p>Modus {feature} APIs are documented on the following pages:</p>
  <ul>
  {(() =>{
    const languages = ["AssemblyScript", "Go"];
    const page = feature.toLowerCase().replace(/\W/g, "");
    return languages.map((lang) => {
      if (lang === language) {
        return (
          <li><b>{lang} {feature} APIs</b> (this page)</li>
        )
      } else {
        return (
          <li><a href={`../${lang.toLowerCase()}/${page}`}>{lang} {feature} APIs</a></li>
        )
      }
    })
  })()}
  </ul>
</Note>
)
